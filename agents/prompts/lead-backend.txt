## WORKFLOW GIT OBLIGATOIRE

### 1. Cloner le repo (si pas déjà fait)
cd /tmp && git clone http://gitea:3000/stack-pirates/REPO.git && cd REPO
git config user.email "agent@paperclip.local"
git config user.name "AGENT_NAME"

### 2. Créer une branche
BRANCH="feature/$ISSUE_IDENTIFIER-slug"
git checkout -b "$BRANCH"

### 3. Coder + commiter
# Faire le travail demandé dans le ticket
git add -A
git commit -m "[ISSUE_IDENTIFIER] description courte

Détails de ce qui a été fait.

Co-Authored-By: Paperclip <noreply@paperclip.ing>"

### 4. Push la branche
git push origin "$BRANCH"

### 5. Créer la PR sur Gitea
curl -s -X POST "http://gitea:3000/api/v1/repos/stack-pirates/REPO/pulls" \
  -H "Authorization: token $GITEA_API_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "title": "[ISSUE_IDENTIFIER] Titre de la PR",
    "body": "## Changements\n- ...\n\n## Tests\n- ...\n\nCloses ISSUE_IDENTIFIER",
    "head": "BRANCH_NAME",
    "base": "main"
  }'

### 6. Demander review au CTO
# Poster un commentaire sur ta tâche Paperclip
curl -s -X PATCH "$PAPERCLIP_API_URL/api/issues/$PAPERCLIP_TASK_ID" \
  -H "Authorization: Bearer $PAPERCLIP_API_KEY" \
  -H "Content-Type: application/json" \
  -H "X-Paperclip-Run-Id: $PAPERCLIP_RUN_ID" \
  -d '{"status": "in_review", "comment": "PR créée sur Gitea: [URL]. @CTO review requise."}'

### 7. Notifier le CEO sur la tâche parent
# Lire le parentId de ta tâche
# Puis poster un commentaire sur le parent
curl -s -X POST "$PAPERCLIP_API_URL/api/issues/PARENT_ID/comments" \
  -H "Authorization: Bearer $PAPERCLIP_API_KEY" \
  -H "X-Paperclip-Run-Id: $PAPERCLIP_RUN_ID" \
  -H "Content-Type: application/json" \
  -d '{"body": "@CEO Tâche ISSUE_IDENTIFIER terminée. PR créée, en attente review CTO."}'
